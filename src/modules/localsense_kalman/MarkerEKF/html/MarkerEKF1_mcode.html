1function [xa_apo,Pa_apo]= MarkerEKF(dt,z,q_a,q_v,q_x,r_a,r_x)
  2%输出
  3% %xa_apo状态变量最优估计，6维，分别为惯性坐标系下的X轴与Y轴的加速度，速度以及位置
  4% ax：惯性坐标系下的X轴加速度
  5% ay：惯性坐标系下的Y轴加速度
  6% vx：惯性坐标系下的X轴速度
  7% vy：惯性坐标系下的Y轴速度
  8% x：惯性坐标系下的X轴坐标
  9% y：惯性坐标系下的Y轴坐标
 10%%Pa_apo协方差矩阵最优估计
 11
 12%输入
 13%persitent x_apo，上一时刻的状态的最优估计
 14%6维，分别为惯性坐标系下的X轴与Y轴的加速度，速度以及位置
 15
 16%persitent p_apo，上一时刻的协方差的最优估计
 17
 18%Z，测量变量，4维
 19% axz：惯性坐标系下的X轴加速度测量值
 20% ayz：惯性坐标系下的Y轴加速度测量值
 21% xz：惯性坐标系下的X轴位置测量值
 22% yz：惯性坐标系下的Y轴位置测量值
 23%dt: dt 积分的步长
 24
 25%% init
 26persistent x_apo
 27if(isempty(x_apo))
 28    x_apo=single([0;0;0;0;10.0;10.0]); 
 29end
 30
 31%给定初始协方差矩阵，迭代过程会收敛%
 32persistent P_apo
 33if(isempty(P_apo))
 34    P_apo = single(ones(6));
 35end
 36
 37ax=x_apo(1);% ax：惯性坐标系下的X轴加速度
 38ay=x_apo(2);% ay：惯性坐标系下的Y轴加速度
 39vx=x_apo(3);% vx：惯性坐标系下的X轴速度
 40vy=x_apo(4);% vy：惯性坐标系下的Y轴速度
 41x=x_apo(5);% x：惯性坐标系下的X轴坐标
 42y=x_apo(6);% y：惯性坐标系下的Y轴坐标
 43
 44%% prediction section
 45%公式1
 46%X(k|k-1)=A X(k-1|k-1)+B U(k) ……….. (1)
 47%系统控制变量的输入为U=0
 48% ax(k|k-1)=ax(k-1|k-1)
 49% ay(k|k-1)=ay(k-1|k-1)
 50% vx(k|k-1)=vx(k-1|k-1)+ax(k-1|k-1)*dt;
 51% vy(k|k-1)=vy(k-1|k-1)+ay(k-1|k-1)*dt;
 52% x(k|k-1)=x(k-1|k-1)+vx(k-1|k-1)*dt+1/2*ax(k-1|k-1)*dt^2;
 53% y(k|k-1)=y(k-1|k-1)+vy(k-1|k-1)*dt+1/2*ay(k-1|k-1)*dt^2;
 54%A=[1 0 0 0 0 0;
 55% 0 1 0 0 0 0;
 56% dt 0 1 0 0 0;
 57% 0 dt 0 1 0 0;
 58% 0 0 dt 0 1 0;
 59% 0 0 0 dt 0 1;
 60% ]
 61ak =[ax;ay];
 62vk =[vx;vy] + dt*ak;
 63rk=[x;y]+[vx;vy]*dt+1/2*ak*dt^2;
 64% rk=[x;y]+dt*vk;
 65x_apr=[ak;vk;rk];%x_apr当前时刻的状态估计
 66
 67%公式2 更新协方差矩阵的预测
 68% P(k|k-1)=A P(k-1|k-1) A’+Q
 69% A=[1 0 0 0 0 0;
 70%    0 1 0 0 0 0;
 71%    dt 0 1 0 0 0;
 72%    0 dt 0 1 0 0;
 73%    0 0 dt 0 1 0;
 74%    0 0 0 dt 0 1;];
 75A=[1 0 0 0 0 0;
 76   0 1 0 0 0 0;
 77   dt 0 1 0 0 0;
 78   0 dt 0 1 0 0;
 79   1/2*dt^2 0 dt 0 1 0;
 80   0 1/2*dt^2 0 dt 0 1;];
 81
 82persistent Q
 83if (isempty(Q))
 84    Q=diag([ q_a,q_a,q_v,q_v,q_x,q_x]);
 85end
 86
 87P_apr=A*P_apo*A'+Q;
 88
 89%% update
 90%先公式4，再公式3 
 91% Kg(k)= P(k|k-1) H’ [ (H P(k|k-1) H’ + R)]^-1 
 92%H 为测量矩阵
 93H=[1,0,0,0,0,0;
 94    0,1,0,0,0,0;
 95    0,0,0,0,1,0;
 96    0,0,0,0,0,1];
 97
 98persistent R
 99if (isempty(R))
100    R=diag([ r_a,r_a,r_x,r_x]);
101end
102 S_k=H*P_apr*H';
103 S_k = S_k+ R;
104 K_k=(P_apr*H'/(S_k));
105%  
106%  公式3，更新最优状态估计
107% X(k|k)= X(k|k-1)+Kg(k) (Z(k)-H X(k|k-1)) ……… (3)
108y_k=z-H*x_apr;
109x_apo=x_apr+K_k*y_k;
110%  
111%  公式5，更新协方差矩阵估计
112% P(k|k)=（I-Kg(k) H）P(k|k-1)
113P_apo=(eye(6)-K_k*H)*P_apr;
114xa_apo=x_apo;
115Pa_apo=P_apo;
